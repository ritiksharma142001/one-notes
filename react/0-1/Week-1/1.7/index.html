let arr = [
    {

// Objects of Objects
let students = {
  ritik: {
    name: "Ritik",
    marks: 98
  },
  ronit: {
    name: "Ronit",
    marks: 88
  },
  rajat: {
    name: "Rajat",
    marks: 68
  }
};
// cannot use .forEach() directly on an object, but you can use Object.values() or Object.entries() to iterate:
Object.values(students).forEach(student => {
  console.log(student.marks);
});

// OR
// What is Object.entries()?
// It converts an object to an array of [key, value] pairs.
Object.entries(students).forEach(([key, student]) => {
  console.log(key, student.name, student.marks);
});


// Or

for (let key in students) {
  console.log(students[key].marks);
}


console.log(Object.entries(students));
// [
//   ['ritik', { name: 'Ritik', marks: 98 }],
//   ['ronit', { name: 'Ronit', marks: 88 }],
//   ['rajat', { name: 'Rajat', marks: 68 }]
// ]

// Object of Arrays
// It means that each key in the object has an array as its value.
let studentMarks = {
  ritik: [95, 98, 99],
  ronit: [85, 88, 90],
  rajat: [65, 68, 70]
};
Object.entries(studentMarks).forEach(([name, marks]) => {
  console.log(`${name}: ${marks.join(", ")}`);
});
// ritik: 95, 98, 99
// ronit: 85, 88, 90
// rajat: 65, 68, 70

for (let name in studentMarks) {
  console.log(name, studentMarks[name]);
}
// ritik (3) [95, 98, 99]
// ronit (3) [85, 88, 90]
// rajat (3) [65, 68, 70]

code-7 splice()​

  • used to add and remove things from a same array
const a = [5, 3, 1, 8, 6, 4];
a.splice(0, 3) // start index, how many element need to delete
console.log(a) // [8, 6, 4]

const a = [5, 3, 1, 8, 6, 4];
a.splice(2, 2, "ritik") 
console.log(a) // [5, 3, 'ritik', 6, 4]

const a = [5, 3, 1, 8, 6, 4];
a.splice(2, 2, "ritik", "eshu", "mohit", "rakesh")

code-8 filter()​

  • return new array
  • filter from array
const a = [5, 3, 1, 8, 6, 4];
const b = a.filter((value, index, array)=> value%2 != 0)
console.log(b) // [5, 3, 1]
console.log(a) // [5, 3, 1, 8, 6, 4]

code-9 map()​

  • return new array
  • to modify array
const a = [5, 3, 1, 8, 6, 4];
const b = a.map((value, index, array)=>value*2)

console.log(a) // [5, 3, 1, 8, 6, 4]

code-10 chaining methods​

const a = [5, 3, 1, 8, 6, 4];
const b = a
  .filter((value) => value % 2 != 0)
  .map((value, index, array) => value * 2);
console.log(b); // [10, 6, 2]
console.log(a); // [5, 3, 1, 8, 6, 4]

code-13 reduce()​

  • reduce the array to single value
const a = [5, 3, 1, 8, 6, 4];
const b = a.reduce((prev, curr, ind, array)=> prev+curr, 0)
console.log(b); // 27

// maximum in an array
let nums = [2, 3, 5, 3, 6, 8, 6, 5, 3];
let result = nums.reduce((prev, curr) => {
  if (curr > prev) return curr;
  else return prev;
});
console.log(result); // 8

code-15 every()​

  • returns true if every element of array gives true for some function. Else returns false.
const a = [1, 3, 4, 5].every((el) => el % 2 == 0);
console.log(a); // false

const b = [1, 3, 7, 5].every((el) => el % 2 != 0);
console.log(b); // true

code-16 some()​

  • returns true if some elements of arrays give true for some functions. Else return false.
const a = [0, 2, 4, 6].some((el) => el % 2 != 0);
console.log(a); // false

const b = [1, 2, 7, 5].some((el) => el % 2 == 0);
console.log(b); // true

  • expands and iterable into multiple values

let nums = [2, 3, 5, 3, 6, 8, 6, 5, 3];
let newArr = [...nums]
console.log(newArr) // [2, 3, 5, 3, 6, 8, 6, 5, 3]

let chars = [..."hello"]
console.log(chars) // ['h', 'e', 'l', 'l', 'o']

// spread with Object Literals
let data = {
    email: "ritik@gmail.com",

How to loop through objects instead?​

Mathod-1​

for...in (loop through keys)

const person = { name: "Ritik", age: 21 };

for (const key in person) {
    console.log(key, person[key]);
}

Method-2​

Object.keys() + for...of

const person = { name: "Ritik", age: 21 };
const keys = Object.keys(person);  // ["name", "age"]

for (const key of keys) {
    console.log(key, person[key]);
}

Method-3​

Object.entries() for key + value together

const person = { name: "Ritik", age: 21 };

for (const [key, value] of Object.entries(person)) {
    console.log(`${key}: ${value}`);
}

String Methods​

  • strings are immutable in js.
  • no changes can be made to Strings.
  • whenever we do try to make a change, a new string is created and old one remains same.

console.log(msg.trim()) // hello

let str = "heLLO woRld"

console.log(str.toUpperCase()) // HELLO WORLD

caode-3 indexOf()​

  • returns the first index of occurrence of some value in string or gives -1 if not found.
