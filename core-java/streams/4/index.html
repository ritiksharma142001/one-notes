import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class FlatMapExample {
    public static void main(String[] args) {
        List<List<String>> nestedList = Arrays.asList(
            Arrays.asList("Apple", "Banana"),
            Arrays.asList("Orange", "Mango"),
            Arrays.asList("Grapes", "Pineapple")
        );

        // Using map() - Returns Stream<List<String>>
        List<List<String>> mappedList = nestedList.stream()
                .map(list -> list)
                .collect(Collectors.toList());

        System.out.println("Using map(): " + mappedList);

        // Using flatMap() - Returns Stream<String>
        List<String> flatMappedList = nestedList.stream()
                .flatMap(list -> list.stream())
                .collect(Collectors.toList());

        System.out.println("Using flatMap(): " + flatMappedList);
    }
}

// Using map(): [[Apple, Banana], [Orange, Mango], [Grapes, Pineapple]]

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class StringFlatMapExample {
    public static void main(String[] args) {
        List<String> sentences = Arrays.asList("Hello World", "Java Streams", "Flat Map Example");

ðŸ”¹ forEachOrdered() in Java Streamsâ€‹

  • forEachOrdered(Consumer<? super T> action) is a terminal operation in Java Streams.
  • It ensures elements are processed in the encounter order (original order of elements).
  • Especially useful with parallel streams when order consistency matters.

import java.util.*;
import java.util.stream.Stream;

public class StreamForEachOrderedExample {
    public static void main(String[] args) {
        // Example 1: Print elements of integer list in original order
        List<Integer> intList = Arrays.asList(10, 19, 20, 1, 2);
        System.out.println("Integer List:");
        intList.stream().forEachOrdered(System.out::println);

        System.out.println(); // Line break

        // Example 2: Print elements of string list in original order
        List<String> strList = Arrays.asList("GFG", "Geeks", "for", "GeeksforGeeks");
        System.out.println("String List:");
        strList.stream().forEachOrdered(System.out::println);

        System.out.println(); // Line break

        // Example 3: Print characters at index 2 of string list in original order
        System.out.println("Characters at Index 2:");
        strList.stream()
               .filter(str -> str.length() > 2) // Ensure index 2 exists
               .flatMap(str -> Stream.of(str.charAt(2))) // Extract the character at index 2
               .forEachOrdered(System.out::println);
    }
}

  • âœ… Supports lambda expressions for cleaner, functional-style code.

  • âŒ Cannot use return, break, or continue inside the lambda.
  • âœ… Works only with collections (like List, Set), not arrays.

  • âœ… Uses multiple threads for faster execution, taking advantage of multi-core processors.
  • âŒ Order is not guaranteed (elements may print in any order).
  • âœ… Typically used for large data sets where speed is a priority over order.

ðŸ”¹ When to Use Which?
âœ… Use forEach loop when you need to modify external variables or return/exit the loop early.

âœ… Use stream().forEach() for simple iteration when performance is not a concern.

âœ… Use parallelStream().forEach() for large datasets where speed matters, but order is not a concern.

âš ï¸ Caution: Using parallelStream() can lead to non-deterministic results (output order may vary) due to the multi-threaded nature of the operation.
